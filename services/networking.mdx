# Networking Services

This document describes the networking services: VPC, Load Balancing, Cloud NAT, and related networking components.

## Virtual Private Cloud (VPC)

### Overview

VPC provides isolated network environments for your resources.

### Configuration

```hcl
resource "google_compute_network" "main" {
  name                    = "main-vpc"
  auto_create_subnetworks = false
  routing_mode            = "REGIONAL"
}

resource "google_compute_subnetwork" "main" {
  name          = "main-subnet"
  ip_cidr_range = "10.0.0.0/24"
  region        = var.region
  network       = google_compute_network.main.id

  private_ip_google_access = true

  log_config {
    aggregation_interval = "INTERVAL_5_SEC"
    flow_sampling        = 0.5
    metadata             = "INCLUDE_ALL_METADATA"
  }
}
```

## Load Balancing

### HTTP(S) Load Balancer

```hcl
resource "google_compute_backend_service" "web" {
  name                  = "web-backend"
  protocol              = "HTTP"
  port_name             = "http"
  load_balancing_scheme = "EXTERNAL_MANAGED"
  timeout_sec           = 30

  backend {
    group = google_compute_instance_group.web.id
  }

  health_checks = [google_compute_health_check.web.id]
}

resource "google_compute_url_map" "web" {
  name            = "web-url-map"
  default_service = google_compute_backend_service.web.id
}

resource "google_compute_target_https_proxy" "web" {
  name             = "web-https-proxy"
  url_map          = google_compute_url_map.web.id
  ssl_certificates = [google_compute_managed_ssl_certificate.web.id]
}

resource "google_compute_global_forwarding_rule" "web" {
  name       = "web-forwarding-rule"
  target     = google_compute_target_https_proxy.web.id
  port_range = "443"
  ip_address = google_compute_global_address.web.address
}
```

### Internal Load Balancer

```hcl
resource "google_compute_region_backend_service" "internal" {
  name                  = "internal-backend"
  protocol              = "TCP"
  load_balancing_scheme = "INTERNAL"
  region                = var.region

  backend {
    group = google_compute_instance_group.internal.id
  }

  health_checks = [google_compute_health_check.internal.id]
}
```

## Cloud NAT

### Configuration

```hcl
resource "google_compute_router" "nat_router" {
  name    = "nat-router"
  region  = var.region
  network = google_compute_network.main.id
}

resource "google_compute_router_nat" "nat" {
  name                               = "nat-gateway"
  router                             = google_compute_router.nat_router.name
  region                             = var.region
  nat_ip_allocate_option             = "AUTO_ONLY"
  source_subnetwork_ip_ranges_to_nat = "ALL_SUBNETWORKS_ALL_IP_RANGES"

  log_config {
    enable = true
    filter = "ERRORS_ONLY"
  }
}
```

## Firewall Rules

### Configuration

```hcl
resource "google_compute_firewall" "allow_http" {
  name    = "allow-http"
  network = google_compute_network.main.name

  allow {
    protocol = "tcp"
    ports    = ["80", "443"]
  }

  source_ranges = ["0.0.0.0/0"]
  target_tags   = ["http-server"]
}

resource "google_compute_firewall" "allow_internal" {
  name    = "allow-internal"
  network = google_compute_network.main.name

  allow {
    protocol = "tcp"
    ports    = ["0-65535"]
  }

  source_ranges = ["10.0.0.0/8"]
}
```

## Cloud DNS

### Configuration

```hcl
resource "google_dns_managed_zone" "main" {
  name        = "main-zone"
  dns_name    = "example.com."
  description = "Main DNS zone"

  visibility = "public"
}

resource "google_dns_record_set" "web" {
  name         = "www.example.com."
  managed_zone = google_dns_managed_zone.main.name
  type         = "A"
  ttl          = 300
  rrdatas      = ["1.2.3.4"]
}
```

## Best Practices

- Use custom mode VPCs
- Enable Private Google Access
- Use firewall rules with tags
- Enable VPC Flow Logs
- Use Cloud NAT for outbound access
